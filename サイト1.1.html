      if (currentFilteredResults.length === 0) {

        container.innerHTML = '<p style="text-align:center; width:100%; padding:20px;">条件に合う口コミが見つかりませんでした。</p>';

        pagiContainer.innerHTML = '';

        return;

      }


      const totalPages = Math.ceil(currentFilteredResults.length / ITEMS_PER_PAGE);

      const start = (currentPage - 1) * ITEMS_PER_PAGE;

      const items = currentFilteredResults.slice(start, start + ITEMS_PER_PAGE);


      container.innerHTML = items.map(p => createCardHtml(p)).join('');


      if (totalPages > 1) {

        let pHTML = '';

        for(let i=1; i<=totalPages; i++){

          pHTML += `<button class="${i === currentPage ? 'active' : ''}" onclick="goToPage(${i})">${i}</button>`;

        }

        pagiContainer.innerHTML = pHTML;

      } else {

        pagiContainer.innerHTML = '';

      }

    }


    function goToPage(p) {

      currentPage = p;

      renderPaginatedResults();

      window.scrollTo(0, 0);

    }


    function createCardHtml(p) {

      let stars = '';

      for(let i=0; i<5; i++) stars += (i < p.stars) ? '★' : '☆';

      const displayTags = (p.tags && p.tags.length > 0) ? p.tags.slice(0, 3).map(t => escapeHTML(t)).join(' / ') : '';

     

      const imgSrc = p.img || `https://placehold.jp/${p.color.replace('#','')}/ffffff/300x200.png?text=No+Image`;


      return `

        <a onclick="showDetail('${p.id}')" class="c-card" style="cursor:pointer;">

          <div class="c-card__img">

            <span class="c-cat-label" style="background:${p.color};">${escapeHTML(p.cat)}</span>

            <img src="${imgSrc}" alt="${escapeHTML(p.title)}">

          </div>

          <div class="c-card__body">

            <h3 class="c-card__title">${escapeHTML(p.title)}</h3>

            <div class="c-rating">${stars}</div>

            <div style="font-size:12px; color:#888; margin-top:5px; white-space:nowrap; overflow:hidden; text-overflow:ellipsis;">

              <i class="fas fa-tag"></i> ${displayTags}

            </div>

          </div>

        </a>`;

    }

    function showDetail(id, pushHistory = true) {

      const item = reviews.find(r => r.id === id);

      if(!item) {

        router('home'); // IDが見つからなければホームに戻す

        return;

      }

      currentDetailId = id;

      router('detail', null, false); // 画面を詳細に切り替え（履歴は残さない）


      // 🔴 タイトルとURLの書き換え

      const title = `${escapeHTML(item.title)}の口コミ・評判 | ミライルーツ`;

      document.title = title;

      if (pushHistory) {

        history.pushState(null, title, `?id=${id}`);

      }


      let stars = '';

      for(let i=0; i<5; i++) stars += (i < item.stars) ? '★' : '☆';


      let urlDisplay = '';

      if(item.url) {

        urlDisplay = `

          <div style="margin-top:20px; padding:15px; border: 2px dashed var(--main-green); border-radius: 10px; background: #fff;">

            <span style="font-weight:bold; color:var(--text-brown);"><i class="fas fa-link"></i> 公式サイト・参考URL：</span><br>

            <a href="${escapeHTML(item.url)}" target="_blank" style="color:var(--main-green); text-decoration:underline; word-break: break-all;">${escapeHTML(item.url)}</a>

          </div>

        `;

      }


      const imgSrc = item.img || `https://placehold.jp/${item.color.replace('#','')}/ffffff/300x200.png?text=No+Image`;

      const tagsHtml = item.tags ? item.tags.map(t => `<span class="tag" style="cursor:default;">${escapeHTML(t)}</span>`).join('') : '';


      document.getElementById('detail-content').innerHTML = `

        <span class="c-cat-label" style="position:static; display:inline-block; background:${item.color}; transform:none; margin-bottom:10px;">${escapeHTML(item.cat)}</span>

        <h2 style="font-size:24px; margin:0 0 10px;">${escapeHTML(item.title)}</h2>

        <div class="c-rating" style="font-size:18px; margin-bottom:15px;">${stars}</div>

        <img src="${imgSrc}" style="width:100%; max-width:500px; border-radius:10px; border:2px solid #ddd; margin-bottom:20px;">


        <div style="display:flex; gap:10px; flex-wrap:wrap; margin-bottom:20px;">

          ${tagsHtml}

        </div>


        <div style="background:var(--bg-cream); padding:20px; border-radius:10px; border:1px solid #eee;">

          <h4 style="margin-top:0; color:var(--text-brown);">良い点・おすすめポイント</h4>

          <p style="white-space: pre-wrap;">${escapeHTML(item.goodPoint)}</p>

        </div>


        ${urlDisplay}

      `;

     

      const commentsQuery = query(collection(db, "comments"), where("itemId", "==", currentDetailId), orderBy("createdAt", "asc"));

      onSnapshot(commentsQuery, (snapshot) => {

        const comments = [];

        snapshot.forEach(doc => comments.push(doc.data()));

        renderComments(comments);

      });

    }


    function renderComments(commentsData) {

      const container = document.getElementById('detail-comments');

      if(commentsData.length === 0) {

        container.innerHTML = '<p style="color:#888; font-size:14px;">まだコメントはありません。最初のコメントを投稿してみましょう！</p>';

      } else {

        container.innerHTML = commentsData.map(c => {

          let s = '';

          for(let i=0; i<5; i++) s += (i < c.star) ? '★' : '☆';

          return `

            <div class="comment-box">

              <div class="comment-header"><span><i class="fas fa-user-circle"></i> ${escapeHTML(c.user)}</span> <span style="color:#f39c12;">${s}</span></div>

              <p style="margin:0; font-size:14px; white-space:pre-wrap;">${escapeHTML(c.text)}</p>

            </div>

          `;

        }).join('');

      }

    }


    async function submitComment(e) {

      e.preventDefault();

      const btn = document.getElementById('btn-submit-comment');

      btn.disabled = true;

      btn.innerText = "送信中...";


      try {

        const star = parseInt(document.getElementById('comment-star').value);

        const text = document.getElementById('comment-text').value;

        const userName = document.getElementById('comment-name').value || 'ゲストユーザー';


        await addDoc(collection(db, "comments"), {

          user: userName,

          star: star,

          text: text,

          createdAt: serverTimestamp()

        });


        document.getElementById('comment-text').value = '';

        document.getElementById('comment-name').value = '';

      } catch (error) {

        console.error("コメント投稿エラー:", error);

        alert("エラーが発生しました。");

      } finally {

        btn.disabled = false;

        btn.innerText = "投稿する";

      }

    }

    async function submitReview(e) {

      e.preventDefault();

      const btn = document.getElementById('btn-submit-review');

      btn.disabled = true;

      btn.innerText = "送信中...";


      try {

        const cat = document.getElementById('input-cat').value;

        const title = document.getElementById('input-title').value;

        const stars = parseInt(document.getElementById('input-stars').value);

        const url = document.getElementById('input-url').value;

        const goodPoint = document.getElementById('input-good').value;


        let tags = [];

        const checkboxes = document.querySelectorAll('#dynamic-form-fields input[type="checkbox"]:checked');

        checkboxes.forEach(cb => tags.push(cb.value));


        await addDoc(collection(db, "reviews"), {

          cat: cat,

          title: title,

          stars: stars,

          goodPoint: goodPoint,

          url: url,

          tags: tags,

          img: '',

          color: getColorForCat(cat),

          createdAt: serverTimestamp()

        });


        alert('投稿ありがとうございます！');

        document.getElementById('input-title').value = '';

        document.getElementById('input-good').value = '';

        document.getElementById('input-url').value = '';

        checkboxes.forEach(cb => cb.checked = false);

       

        router('home'); // 投稿後はホームへ


      } catch (error) {

        console.error("投稿エラー: ", error);

        alert('エラーが発生しました。設定を確認してください。');

      } finally {

        btn.disabled = false;

        btn.innerText = "投稿する";

      }

    }


    function getColorForCat(cat) {

      if(cat === '動画・アニメ') return '#ff7675';

      if(cat === 'アプリ') return '#a29bfe';

      if(cat === '絵本・本') return '#fdcb6e';

      if(cat === '通信・教室') return '#e17055';

      if(cat === 'イベント') return '#74b9ff';

      if(cat === 'ワークシート') return '#55efc4';

      return '#888';

    }
